```bash
get_cache_key() {
    local file="$1"

    # Convert to absolute path
    local abs_path
    if command -v realpath &>/dev/null; then
        abs_path=$(realpath "$file" 2>/dev/null) || abs_path="$file"
    else
        # Fallback
        abs_path="$(pwd)/$file"
    fi

    # Replace / with _ and remove leading /
    echo "$abs_path" | sed 's/^[\/]*//; s/\//_/g'
}
```
